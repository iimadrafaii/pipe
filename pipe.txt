import multiprocessing
import time

def long_running_function(name):
    print(f"Process {name} started")
    time.sleep(5)
    print(f"Process {name} finished")

if __name__ == '__main__':
    # Create the process
    p = multiprocessing.Process(target=long_running_function, args=("A",))
    p.start()

    # Wait for 2 seconds only
    p.join(timeout=2)

    # If thread is still active using is_alive()
    if p.is_alive():
        print("Function timed out! Killing process...")
        p.terminate()  # Forcefully kill the process
        p.join()       # Clean up
    else:
        print("Function finished successfully.")
