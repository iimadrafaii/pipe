from func_timeout import func_timeout, FunctionTimedOut

def long_running_function():
    import time
    time.sleep(5)
    return "Done"

try:
    # Run function with a 2-second timeout
    result = func_timeout(2, long_running_function)
    print(result)
except FunctionTimedOut:
    print("The function took too long and was terminated.")
except Exception as e:
    print(f"Another error occurred: {e}")

import multiprocessing
import time

def long_running_function(name):
    print(f"Process {name} started")
    time.sleep(5)
    print(f"Process {name} finished")

if __name__ == '__main__':
    # Create the process
    p = multiprocessing.Process(target=long_running_function, args=("A",))
    p.start()

    # Wait for 2 seconds only
    p.join(timeout=2)

    # If thread is still active using is_alive()
    if p.is_alive():
        print("Function timed out! Killing process...")
        p.terminate()  # Forcefully kill the process
        p.join()       # Clean up
    else:
        print("Function finished successfully.")
